#include <iostream>
#include <list>
using namespace std;

list<int> intersection(list<int>& setA, list<int>& setB) {
    list<int> result;
    for (int x : setA) {
        for (int y : setB) {
            if (x == y) result.push_back(x);
        }
    }
    return result;
}

list<int> symmetricDifference(list<int>& setA, list<int>& setB) {
    list<int> result;
    for (int x : setA) {
        if (find(setB.begin(), setB.end(), x) == setB.end())
            result.push_back(x);
    }
    for (int y : setB) {
        if (find(setA.begin(), setA.end(), y) == setA.end())
            result.push_back(y);
    }
    return result;
}

int main() {
    list<int> vanilla = {1, 2, 3, 4};
    list<int> butterscotch = {3, 4, 5, 6};

    list<int> both = intersection(vanilla, butterscotch);
    list<int> eitherNotBoth = symmetricDifference(vanilla, butterscotch);

    int totalStudents = 10;
    int neither = totalStudents - vanilla.size() - butterscotch.size() + both.size();

    cout << "Both: ";
    for (int x : both) cout << x << " ";
    cout << endl;

    cout << "Either but not both: ";
    for (int x : eitherNotBoth) cout << x << " ";
    cout << endl;

    cout << "Neither: " << neither << endl;

    return 0;
}
